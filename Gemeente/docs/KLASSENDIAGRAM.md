## ğŸ—ï¸ Architectuur Patroon

**MVC (Model-View-Controller)** met aanvullende lagen:

### Model Layer
- **Eloquent ORM** models
- Database relaties (hasMany, belongsTo)
- Scopes en accessors
- Validatie regels

### Controller Layer
- **Request handling**
- **Business logic** delegatie
- **Response formatting**
- **Middleware integration**

### View Layer
- **Blade templates**
- **Alpine.js** voor interactiviteit
- **Leaflet.js** voor kaarten
- **Component-based** structuur

### Service Layer
- **FileUploadService** - Bestandsverwerking
- Herbruikbare business logic
- Externe API integraties

// User.php
class User extends Authenticatable {
    public function assignedComplaints() {
        return $this->hasMany(Complaint::class, 'assigned_to');
    }
    
    public function notes() {
        return $this->hasMany(Note::class);
    }
    
    public function statusHistories() {
        return $this->hasMany(StatusHistory::class);
    }
}

// Complaint.php
class Complaint extends Model {
    public function attachments() {
        return $this->hasMany(Attachment::class);
    }
    
    public function notes() {
        return $this->hasMany(Note::class);
    }
    
    public function statusHistories() {
        return $this->hasMany(StatusHistory::class);
    }
    
    public function assignedTo() {
        return $this->belongsTo(User::class, 'assigned_to');
    }
}

// Attachment.php
class Attachment extends Model {
    public function complaint() {
        return $this->belongsTo(Complaint::class);
    }
}

// Note.php
class Note extends Model {
    public function complaint() {
        return $this->belongsTo(Complaint::class);
    }
    
    public function user() {
        return $this->belongsTo(User::class);
    }
}

// StatusHistory.php
class StatusHistory extends Model {
    public function complaint() {
        return $this->belongsTo(Complaint::class);
    }
    
    public function user() {
        return $this->belongsTo(User::class);
    }
}
